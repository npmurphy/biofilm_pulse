\newcommand{\drawobjective}[3]{
  % arugment 1 is some unique ider and #2 is the location
  % arugment 3 is the scale

  \pgfmathsetmacro\objtipWidth{7mm * #3}
  \pgfmathsetmacro\objtipHeight{2mm * #3}
  \pgfmathsetmacro\objbaseWidth{15mm * #3}
  \pgfmathsetmacro\objbaseHeight{8mm * #3}

	\node at #2
	[anchor=top side,
		objective,
		yshift=-0.3mm,
		trapezium angle=40,
		minimum width=\objtipWidth,
    minimum height=\objtipHeight,
    draw, ultra thin] (objtip#1) {};
	\node at (objtip#1.south)
	[anchor= top side,
		objective,
		ultra thin,
		trapezium angle=70,
		minimum width=\objbaseWidth,
		minimum height=\objbaseHeight] (objbase#1) {};
}

\newcommand{\drawWilcoDish}[2]{
	\coordinate (centerofglass#1) at #2;

	%% cover slip
	\path [draw] ($(centerofglass#1) + (-1.5, 0)$) --
	($(centerofglass#1) + (+1.5, 0)$);

	\coordinate (dishSW#1) at ($(centerofglass#1) + (-2,0)$);
	\coordinate (dishNW#1) at ($(dishSW#1) + (0, 1.3)$);
	\coordinate (dishSE#1) at ($(centerofglass#1) + (2,0)$);
	\coordinate (dishNE#1) at ($(dishSE#1) + (0, 1.3)$);
	\path[dish] ($(centerofglass#1) + (-1.5, 0)$) -- (dishSW#1) -- (dishNW#1);
	\path[dish] ($(centerofglass#1) + (+1.5,0)$) -- (dishSE#1) -- (dishNE#1);
	% lid
	\coordinate (lidNW#1) at ($(dishNW#1) + (-0.15, 0.1)$);
	\coordinate (lidNE#1) at ($(dishNE#1) +  (0.15, 0.1)$);
	\path[dish] ($(lidNW#1) + (0, -0.8)$) -- (lidNW#1) -- (lidNE#1) -- ($(lidNE#1) + (0, -0.8)$);
}
