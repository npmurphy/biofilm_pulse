\documentclass[tikz]{standalone}

\usepackage{tikz}
\usetikzlibrary{shapes.geometric}
\usetikzlibrary{%
    calc,%
    decorations.pathmorphing,%
    fadings,%
    shadings%
}

\renewcommand*{\familydefault}{\sfdefault}
    
\begin{document}

\tikzstyle{objective}=[
trapezium,
left color=gray!80!black,
right color=gray!20!white,
draw, thick]
%trapezium stretches,
\tikzstyle{dish}=[draw = blue, very thick]

\begin{tikzpicture}

	\node at (1.60, 0)
    [anchor= bottom left corner,
      objective,
      trapezium angle=70,
      minimum width=15mm,
      minimum height=8mm] (objbase) {};

	\node at (objbase.north west)
    [anchor= bottom left corner,
      objective,
      yshift=-0.3mm,
      trapezium angle=40,
      minimum width=7mm,
      minimum height=2mm, draw, thick] (objtip) {};
  
  % \draw[white,fill=brown!50!white,path fading=east]
  %   (objtip.top side) rectangle +(3,1);
  \node at (objtip.top side)
       [rectangle,
         minimum width=3cm,
         minimum height=1cm,
         anchor =south west,
         fill=brown!50!white, path fading=east] (agar) {};

  %% cover slip
  \path [draw] 
    ($(objtip.top side) + (-1, 0)$) --
	  ($(objtip.top side) + (+1, 0)$);
  
  \coordinate (dishSW) at ($(objtip.top side) + (-3,0)$);
  \coordinate (dishNW) at ($(dishSW) + (0, 1.5)$);
  \coordinate (dishSE) at ($(objtip.top side) + (3,0)$);
  \coordinate (dishNE) at ($(dishSE) + (0, 1.5)$);

  \path[dish] 
    ($(objtip.top side) + (-1, 0)$) -- (dishSW) -- (dishNW);
  \path[dish]
     ($(objtip.top side) + (+1,0)$) -- (dishSE) -- (dishNE);

  % lid
  \path[dish]
    ($(dishNW) + (-0.1, -0.5)$) -- ++ (0, 0.7) -- ++ (6.2, 0) -- ++ (0, -0.7);

  % biofilm
  \path[draw=brown, thin, fill=yellow]
    ($(agar.south west) + (0, 0.01)$)
                   -- ++ (-0.3, 0.0)
      decorate [decoration={random steps,segment length=0.05cm,amplitude=.02cm}]
      { -- ++ (0, 0.7) -- ++ (+0.3, 0)}
       -- ($(agar.south west) + (0, 0.01)$);
  

\end{tikzpicture}
\end{document}
