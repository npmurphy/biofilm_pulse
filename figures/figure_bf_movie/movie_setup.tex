  \newcommand{\drawobjective}[2]{
    % arugment 1 is some unique ider and #2 is the location
    \node at #2 
        [anchor=top side,
          objective,
          yshift=-0.3mm,
          trapezium angle=40,
          minimum width=3.5mm,
          minimum height=1mm, draw, ultra thin] (objtip#1) {};
    \node at (objtip#1.south)
        [anchor= top side,
          objective,
          ultra thin,
          trapezium angle=70,
          minimum width=7.5mm,
          minimum height=4mm] (objbase#1) {};
  }

  \newcommand{\drawWilcoDish}[2]{

    \coordinate (dishNE#1) at ($(dishSE#1) + (0, 1.3)$);
    \path[dish] ($(centerofglass#1) + (-1, 0)$) -- (dishSW#1) -- (dishNW#1);
    \path[dish] ($(centerofglass#1) + (+1,0)$) -- (dishSE#1) -- (dishNE#1);

    \path[dish] ($(lidNW#1) + (0, -0.8)$) -- (lidNW#1) -- (lidNE#1) -- ($(lidNE#1) + (0, -0.8)$);
  }
